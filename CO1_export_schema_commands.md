  OWNER=SCHEMA1 \
  FILE=CO1_APP1_APP1DB1_SCHEMA1_ddl.dmp \
  EXCLUDE=TABLE:"IN ('TABLE1','TABLE5','TABLE6','TABLE10')" \
  TABLES=SCHEMA1.TABLE2, SCHEMA1.TABLE3, SCHEMA1.TABLE4, SCHEMA1.TABLE7, SCHEMA1.TABLE8, SCHEMA1.TABLE9 \
  QUERY="SCHEMA1.TABLE2:'WHERE STATE_CD = ''LA'' '" \
  QUERY="SCHEMA1.TABLE7:'WHERE STATE_CD = ''LA'' '"
